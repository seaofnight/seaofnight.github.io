machine create

{ NODE_NAME=provision VRAM=1024 VCPU=2 OSTYPE=linux OSVARIANT=virtio26 DISK="path=/data/kvm_disk/container-linux/container-linux1.qcow2,format=qcow2,bus=virtio" NETWORK="network:ovs-network,mac=52:54:00:fe:b3:c0" OUTPUT=/var/lib/libvirt/container-linux/container-linux1/domain.xml

virt-install 
    --connect qemu:///system \
    --import \
    --name ${NODE_NAME} \
    --ram ${VRAM} \ 
    --vcpus ${VCPU} \
    --os-type=${OSTYPE} \
    --os-variant=${OSVARIANT} \
    --disk ${DISK} \
    --network ${NETWORK}
    --vnc --noautoconsole \
    --print-xml &gt; ${OUTPUT}

sed -ie 's|type="kvm"|type="kvm" xmlns:qemu="http://libvirt.org/schemas/domain/qemu/1.0"|' "${OUTPUT}"
sed -i "/&lt;\/devices&gt;/a &lt;qemu:commandline&gt;\n  &lt;qemu:arg value='-fw_cfg'/&gt;\n  &lt;qemu:arg value='name=opt/com.coreos/config,file=${ignition_file}'/&gt;\n&lt;/qemu:commandline&gt;" "${OUTPUT}"

virsh define ${OUTPUT}
virsh start ${NODE_NAME}

}
